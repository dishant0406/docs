## Create the client once

```ts
import { WhyOps } from '@whyops/sdk';

export const whyops = new WhyOps({
  apiKey: process.env.WHYOPS_API_KEY!,
  agentName: 'customer-support-agent',
  agentMetadata: {
    systemPrompt: 'You are a precise customer support assistant.',
    description: 'Handles support, billing, and order status flows.',
    tools: [
      {
        name: 'search_orders',
        description: 'Look up order state by ID',
        inputSchema: JSON.stringify({
          type: 'object',
          properties: { orderId: { type: 'string' } },
          required: ['orderId'],
        }),
        outputSchema: JSON.stringify({
          type: 'object',
          properties: { status: { type: 'string' } },
        }),
      },
    ],
  },
});
```

<Callout type="tip" title="Agent init behavior">
  `trace()` methods auto-register the agent on first use. Call `await whyops.initAgent()` during boot if you want startup-time validation or deploy-time visibility.
</Callout>

## Choose the operating mode

<Steps>
  <Step title="Proxy mode">
    Patch an OpenAI or Anthropic client when you want automatic capture of prompts, responses, tool calls, embeddings, and trace continuity.
  </Step>
  <Step title="Manual events mode">
    Add `trace()` events when you need app-side tool timings, retries, orchestration state, or failures the proxy cannot see.
  </Step>
  <Step title="Hybrid mode">
    Use both together when you want provider traffic from the proxy and richer runtime visibility from your application code.
  </Step>
</Steps>

## Proxy mode

<Tabs>
  <Tab title="OpenAI helper">
    ```ts
    import OpenAI from 'openai';
    import { whyops } from './whyops';

    const openai = whyops.openai(
      new OpenAI({ apiKey: process.env.OPENAI_API_KEY }),
    );

    const response = await openai.chat.completions.create({
      model: 'gpt-4o',
      messages: [{ role: 'user', content: 'Where is order 123?' }],
    });
    ```
  </Tab>
  <Tab title="Anthropic helper">
    ```ts
    import Anthropic from '@anthropic-ai/sdk';
    import { whyops } from './whyops';

    const anthropic = whyops.anthropic(
      new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY }),
    );

    const message = await anthropic.messages.create({
      model: 'claude-3-5-sonnet-20241022',
      max_tokens: 800,
      messages: [{ role: 'user', content: 'Summarize this incident.' }],
    });
    ```
  </Tab>
  <Tab title="What the helper changes">
    ```ts
    // whyops.openai(client) mutates the client in place:
    // - baseURL -> https://proxy.whyops.com
    // - apiKey -> WHYOPS_API_KEY
    // - Authorization -> Bearer WHYOPS_API_KEY
    // - X-Agent-Name -> your stable agent name
    //
    // Keep a separate unpatched provider client if you still need direct calls.
    ```
  </Tab>
</Tabs>

## Self-hosted overrides

```ts
const whyops = new WhyOps({
  apiKey: process.env.WHYOPS_API_KEY!,
  agentName: 'customer-support-agent',
  agentMetadata: { systemPrompt: 'You are helpful.', tools: [] },
  proxyBaseUrl: 'https://proxy.internal.example.com',
  analyseBaseUrl: 'https://analyse.internal.example.com/api',
});
```
